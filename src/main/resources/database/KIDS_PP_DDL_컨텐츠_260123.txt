-- 대국민포털_컨텐츠기본
CREATE TABLE KIDS_OWN.TB_PP_M_CONTS
(
	conts_sn     VARCHAR(10)  NOT NULL, -- 콘텐츠일련번호
    conts_ver_no NUMBER(12,0) NOT NULL, -- 콘텐츠버전일련번호
    conts_ttl    VARCHAR(200) NOT NULL, -- 콘텐츠제목
	conts_cn     TEXT         NOT NULL, -- 콘텐츠내용
	conts_use_yn CHAR(1)      NOT NULL, -- 콘텐츠사용여부
	rgtr_id      VARCHAR(40)  NULL,     -- 등록자아이디
	reg_dt       TIMESTAMP    NULL,     -- 등록일시
	mdfr_id      VARCHAR(40)  NULL,     -- 수정자아이디
	mdfcn_dt     TIMESTAMP    NULL      -- 수정일시
);

-- 대국민포털_컨텐츠기본 기본키
CREATE UNIQUE INDEX PK_TB_PP_M_CONTS
    ON KIDS_OWN.TB_PP_M_CONTS
    ( -- 대국민포털_컨텐츠기본
         conts_sn ASC NULLS LAST -- 콘텐츠일련번호
        ,conts_ver_no
    );

-- 대국민포털_컨텐츠기본
ALTER TABLE KIDS_OWN.TB_PP_M_CONTS
	ADD CONSTRAINT PK_TB_PP_M_CONTS
		-- 대국민포털_컨텐츠기본 기본키
	PRIMARY KEY
	USING INDEX PK_TB_PP_M_CONTS
